CREATE OR REPLACE PROCEDURE PROSEDUR
(
  p_USER_NAME  IN VARCHAR2,
  p_email     IN VARCHAR2,
  p_PASSWORD  IN VARCHAR2,
  p_SCORE_POINT_WALLET  IN NUMBER,
  p_AUTHORITY  IN NUMBER,
  p_USER_LEVEL IN NUMBER,
  p_Status    OUT NUMBER
)
AS

BEGIN
    DELETE FROM USERS WHERE USER_NAME=p_USER_NAME;
  BEGIN 
 SELECT COUNT(*) IF  (USER_NAME=p_USER_NAME) FROM COMMENTS
DELETE FROM COMMENTS WHERE USER_NAME=p_USER_NAME;
  END; 
END PROSEDUR;
